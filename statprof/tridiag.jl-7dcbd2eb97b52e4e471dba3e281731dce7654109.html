18 """
19 SymTridiagonal(dv::V, ev::V) where V <: AbstractVector
20
21 Construct a symmetric tridiagonal matrix from the diagonal (`dv`) and first
22 sub/super-diagonal (`ev`), respectively. The result is of type `SymTridiagonal`
23 and provides efficient specialized eigensolvers, but may be converted into a
24 regular matrix with [`convert(Array, _)`](@ref) (or `Array(_)` for short).
25
26 For `SymTridiagonal` block matrices, the elements of `dv` are symmetrized.
27 The argument `ev` is interpreted as the superdiagonal. Blocks from the
28 subdiagonal are (materialized) transpose of the corresponding superdiagonal blocks.
29
30 # Examples
31 ```jldoctest
32 julia> dv = [1, 2, 3, 4]
33 4-element Vector{Int64}:
34 1
35 2
36 3
37 4
38
39 julia> ev = [7, 8, 9]
40 3-element Vector{Int64}:
41 7
42 8
43 9
44
45 julia> SymTridiagonal(dv, ev)
46 4×4 SymTridiagonal{Int64, Vector{Int64}}:
47 1 7 ⋅ ⋅
48 7 2 8 ⋅
49 ⋅ 8 3 9
50 ⋅ ⋅ 9 4
51
52 julia> A = SymTridiagonal(fill([1 2; 3 4], 3), fill([1 2; 3 4], 2));
53
54 julia> A[1,1]
55 2×2 Symmetric{Int64, Matrix{Int64}}:
56 1 2
57 2 4
58

60 2×2 Matrix{Int64}:
61 1 2

222 function dot(x::AbstractVector, S::SymTridiagonal, y::AbstractVector)
223 require_one_based_indexing(x, y)
224 nx, ny = length(x), length(y)
225 (nx == size(S, 1) == ny) || throw(DimensionMismatch("dot"))
226 if nx ≤ 1
227 nx == 0 && return dot(zero(eltype(x)), zero(eltype(S)), zero(eltype(y)))
228 return dot(x[1], S.dv[1], y[1])
229 end
230 dv, ev = S.dv, S.ev
231 @inbounds begin
232 x₀ = x[1]
233 x₊ = x[2]
234 sub = transpose(ev[1])
235 r = dot(adjoint(dv[1])*x₀ + adjoint(sub)*x₊, y[1])
236 for j in 2:nx-1
237 x₋, x₀, x₊ = x₀, x₊, x[j+1]
238 sup, sub = transpose(sub), transpose(ev[j])
239 r += dot(adjoint(sup)*x₋ + adjoint(dv[j])*x₀ + adjoint(sub)*x₊, y[j])
240 end
241 r += dot(adjoint(transpose(sub))*x₀ + adjoint(dv[nx])*x₊, y[nx])
242 end
243 return r
244 end

392 # Implements the determinant using principal minors
393 # a, b, c are assumed to be the subdiagonal, diagonal, and superdiagonal of
394 # a tridiagonal matrix.
395 #Reference:
396 # R. Usmani, "Inversion of a tridiagonal Jacobi matrix",
397 # Linear Algebra and its Applications 212-213 (1994), pp.413-414
398 # doi:10.1016/0024-3795(94)90414-6
399 function det_usmani(a::V, b::V, c::V, shift::Number=0) where {T,V<:AbstractVector{T}}
400 require_one_based_indexing(a, b, c)
401 n = length(b)
402 θa = oneunit(T)+zero(shift)
403 if n == 0
404 return θa
405 end
406 θb = b[1]+shift
407 for i in 2:n
408 θb, θa = (b[i]+shift)*θb - a[i-1]*c[i-1]*θa, θb
409 end
410 return θb
411 end

605 function diag(M::Tridiagonal{T}, n::Integer=0) where T
606 # every branch call similar(..., ::Int) to make sure the
607 # same vector type is returned independent of n
608 if n == 0
609 return copyto!(similar(M.d, length(M.d)), M.d)
610 elseif n == -1
611 return copyto!(similar(M.dl, length(M.dl)), M.dl)
612 elseif n == 1
613 return copyto!(similar(M.du, length(M.du)), M.du)
614 elseif abs(n) <= size(M,1)
615 return fill!(similar(M.d, size(M,1)-abs(n)), zero(T))
616 else
617 throw(ArgumentError(string("requested diagonal, $n, must be at least $(-size(M, 1)) ",
618 "and at most $(size(M, 2)) for an $(size(M, 1))-by-$(size(M, 2)) matrix")))
619 end
620 end

649 @boundscheck checkbounds(A, i, j)

697 function tril!(M::Tridiagonal{T}, k::Integer=0) where T
698 n = length(M.d)
699 if !(-n - 1 <= k <= n - 1)
700 throw(ArgumentError(string("the requested diagonal, $k, must be at least ",
701 "$(-n - 1) and at most $(n - 1) in an $n-by-$n matrix")))
702 elseif k < -1
703 fill!(M.dl, zero(T))
704 fill!(M.d, zero(T))
705 fill!(M.du, zero(T))
706 elseif k == -1
707 fill!(M.d, zero(T))
708 fill!(M.du, zero(T))
709 elseif k == 0
710 fill!(M.du, zero(T))
711 end
712 return M
713 end

718 throw(ArgumentError(string("the requested diagonal, $k, must be at least ",
719 "$(-n + 1) and at most $(n + 1) in an $n-by-$n matrix")))
720 elseif k > 1

723 fill!(M.du, zero(T))

725 fill!(M.dl, zero(T))
726 fill!(M.d, zero(T))
727 elseif k == 0

773 function Base._sum(A::Tridiagonal, dims::Integer)
774 res = Base.reducedim_initarray(A, dims, zero(eltype(A)))
775 n = length(A.d)
776 if n == 0
777 return res
778 elseif n == 1
779 res[1] = A.d[1]
780 return res
781 end
782 @inbounds begin
783 if dims == 1
784 res[1] = A.dl[1] + A.d[1]
785 for i = 2:n-1
786 res[i] = A.dl[i] + A.d[i] + A.du[i-1]
787 end
788 res[n] = A.d[n] + A.du[n-1]
789 elseif dims == 2
790 res[1] = A.d[1] + A.du[1]
791 for i = 2:n-1
792 res[i] = A.dl[i-1] + A.d[i] + A.du[i]
793 end
794 res[n] = A.dl[n-1] + A.d[n]
795 elseif dims >= 3
796 for i = 1:n-1
797 res[i,i+1] = A.du[i]
798 res[i,i] = A.d[i]
799 res[i+1,i] = A.dl[i]
800 end
801 res[n,n] = A.d[n]
802 end
803 end
804 res
805 end

807 function Base._sum(A::SymTridiagonal, dims::Integer)
808 res = Base.reducedim_initarray(A, dims, zero(eltype(A)))
809 n = length(A.dv)
810 if n == 0
811 return res
812 elseif n == 1
813 res[1] = A.dv[1]
814 return res
815 end
816 @inbounds begin
817 if dims == 1
818 res[1] = transpose(A.ev[1]) + symmetric(A.dv[1], :U)
819 for i = 2:n-1
820 res[i] = transpose(A.ev[i]) + symmetric(A.dv[i], :U) + A.ev[i-1]
821 end
822 res[n] = symmetric(A.dv[n], :U) + A.ev[n-1]
823 elseif dims == 2
824 res[1] = symmetric(A.dv[1], :U) + A.ev[1]
825 for i = 2:n-1
826 res[i] = transpose(A.ev[i-1]) + symmetric(A.dv[i], :U) + A.ev[i]
827 end
828 res[n] = transpose(A.ev[n-1]) + symmetric(A.dv[n], :U)
829 elseif dims >= 3
830 for i = 1:n-1
831 res[i,i+1] = A.ev[i]
832 res[i,i] = symmetric(A.dv[i], :U)
833 res[i+1,i] = transpose(A.ev[i])
834 end
835 res[n,n] = symmetric(A.dv[n], :U)
836 end
837 end
838 res
839 end

841 function dot(x::AbstractVector, A::Tridiagonal, y::AbstractVector)
842 require_one_based_indexing(x, y)
843 nx, ny = length(x), length(y)
844 (nx == size(A, 1) == ny) || throw(DimensionMismatch())
845 if nx ≤ 1
846 nx == 0 && return dot(zero(eltype(x)), zero(eltype(A)), zero(eltype(y)))
847 return dot(x[1], A.d[1], y[1])
848 end
849 @inbounds begin
850 x₀ = x[1]
851 x₊ = x[2]
852 dl, d, du = A.dl, A.d, A.du
853 r = dot(adjoint(d[1])*x₀ + adjoint(dl[1])*x₊, y[1])
854 for j in 2:nx-1
855 x₋, x₀, x₊ = x₀, x₊, x[j+1]
856 r += dot(adjoint(du[j-1])*x₋ + adjoint(d[j])*x₀ + adjoint(dl[j])*x₊, y[j])
857 end
858 r += dot(adjoint(du[nx-1])*x₀ + adjoint(d[nx])*x₊, y[nx])
859 end
860 return r
861 end
